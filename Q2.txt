Q2. Write a program that grades students based on their marks.


let grade = 90;

if (grade >= 90){
    console.log("A Grade");
}else if(grade >= 70 && grade <= 90) {
    console.log("B Grade");
}else if(grade >= 50 && grade <= 70){
    console.log("C Grade");
}else if(grade <= 50){
    console.log("F Grade");
}else{
    console.log("PW Skills")
}
